### 3.3 Quota Refresh Mechanism

#### 3.3.1 Daily External Model Calls
```typescript
// Refresh Logic (based on utils/model-limits.ts)
// Resets daily at 00:00 Beijing Time (UTC+8)
function getTodayString(): string {
  const now = new Date();
  const utc = now.getTime() + now.getTimezoneOffset() * 60000;
  const beijing = new Date(utc + 8 * 3600000);  // Convert to Beijing Time
  return beijing.toISOString().split("T")[0];   // Returns YYYY-MM-DD
}

// Check if reset needed
if (wallet.daily_external_day !== getTodayString()) {
  // Reset daily used count to 0
  wallet.daily_external_used = 0;
  wallet.daily_external_day = getTodayString();
}
```

#### 3.3.2 Monthly Multimodal Quota (Image/Video/Audio)
```typescript
// Refresh Logic (based on services/wallet-supabase.ts)
// Resets on billing anchor day with month-end stickiness

function shouldResetMonthlyQuota(wallet: UserWallet): boolean {
  const now = new Date();
  const resetAt = wallet.monthly_reset_at ? new Date(wallet.monthly_reset_at) : null;

  if (!resetAt) return true;  // Never reset before, needs reset

  // Check if past monthly reset date
  return now >= resetAt;
}

// Monthly reset calculation
function getNextMonthlyResetDate(currentResetAt: Date, anchorDay: number): Date {
  // Uses addCalendarMonths function
  // anchorDay: billing anchor day (e.g., 1st, 15th, end of month)
  // Supports month-end stickiness: if anchor is 31st, Feb adjusts to 28/29th
  return addCalendarMonths(currentResetAt, 1, anchorDay);
}
```

async function deductQuota(userId: string, imageCount: number, videoAudioCount: number) {
  const wallet = await fetchWallet(userId);

  // 1. First deduct from monthly quota (expires)
  const monthlyImageDeduct = Math.min(imageCount, wallet.monthly_image_balance);

  const addonImageDeduct = Math.min(remainingImage, wallet.addon_image_balance);
